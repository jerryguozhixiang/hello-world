
public static void main(String[] args) {
		//用for求100以内的奇数和
		int jSum = 0;//保存奇数和
		int oSum = 0;//保存偶数和
		for(int i = 0; i <= 100; i++){
			if(i%2 != 0){
				jSum += i;
			}else{
				oSum += i;
			}
		}
		System.out.println("奇数的和为：" + jSum);
		System.out.println("偶数的和为：" + oSum);
	}
